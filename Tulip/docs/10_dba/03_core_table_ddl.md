## 1. 문서 개요

본 문서는 `01_data_modeling_principles.md`의 원칙과 `02_master_erd.md`의 ERD를 바탕으로 **핵심 테이블 32개**의 실행 가능한 PostgreSQL DDL을 정의한다.

### 1.1 사전 준비

```sql
-- 확장 활성화
CREATE EXTENSION IF NOT EXISTS pgcrypto;
CREATE EXTENSION IF NOT EXISTS pg_trgm;
CREATE EXTENSION IF NOT EXISTS unaccent;
CREATE EXTENSION IF NOT EXISTS btree_gin;
CREATE EXTENSION IF NOT EXISTS pg_stat_statements;

-- 스키마
CREATE SCHEMA IF NOT EXISTS tulip;
SET search_path = tulip, public;

-- timezone·collation
SET TIMEZONE = 'UTC';
-- 데이터베이스 생성 시: ENCODING 'UTF8' LC_COLLATE 'ko-x-icu' LC_CTYPE 'ko-x-icu'
```

### 1.2 공통 컬럼 매크로 (참조용)

모든 업무 테이블은 다음을 포함한다. (DDL 중복 줄이기 위해 본 문서에서는 명시 표기.)

```
id              BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
tenant_id       BIGINT NOT NULL,
library_id      BIGINT NOT NULL DEFAULT 0,
created_at      TIMESTAMPTZ NOT NULL DEFAULT now(),
created_by      BIGINT,
updated_at      TIMESTAMPTZ NOT NULL DEFAULT now(),
updated_by      BIGINT,
deleted_at      TIMESTAMPTZ,
version         INTEGER NOT NULL DEFAULT 0
```

### 1.3 공통 트리거 함수

```sql
CREATE OR REPLACE FUNCTION fn_set_updated_at()
RETURNS TRIGGER AS $$
BEGIN
  NEW.updated_at := now();
  NEW.version    := COALESCE(OLD.version, 0) + 1;
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE OR REPLACE FUNCTION fn_current_tenant_id()
RETURNS BIGINT AS $$
BEGIN
  RETURN current_setting('app.tenant_id', true)::BIGINT;
EXCEPTION WHEN OTHERS THEN
  RETURN NULL;
END;
$$ LANGUAGE plpgsql STABLE;
```

### 3.2 `tlp_cat_marc_field` — MARC 필드 (가변·반복·서브필드)

```sql
CREATE TABLE tlp_cat_marc_field (
  id              BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  tenant_id       BIGINT NOT NULL,
  bibliography_id BIGINT NOT NULL REFERENCES tlp_cat_bibliography(id) ON DELETE CASCADE,
  tag             VARCHAR(3) NOT NULL,             -- 020, 245, 700, 008 ...
  ind1            CHAR(1),
  ind2            CHAR(1),
  occurrence_no   SMALLINT NOT NULL DEFAULT 1,     -- 반복필드 순번
  field_type      CHAR(1) NOT NULL DEFAULT 'D'
                    CHECK (field_type IN ('C','D')),  -- C=Control(00X) / D=Data
  control_value   TEXT,                            -- 제어필드 직접값
  subfields       JSONB NOT NULL DEFAULT '[]'::jsonb,
                  -- 예: [{"code":"a","value":"홍길동전"},{"code":"d","value":"홍길동"}]
  CONSTRAINT uk_marc_field UNIQUE (bibliography_id, tag, occurrence_no),
  CONSTRAINT ck_marc_field_value CHECK (
    (field_type = 'C' AND control_value IS NOT NULL)
    OR (field_type = 'D' AND jsonb_typeof(subfields) = 'array')
  )
);

CREATE INDEX ix_marc_bib_tag ON tlp_cat_marc_field(bibliography_id, tag);
CREATE INDEX gx_marc_subfields ON tlp_cat_marc_field USING GIN (subfields jsonb_path_ops);
-- 서브필드 a 표현식 인덱스 (예: 245$a)
CREATE INDEX fx_marc_245a ON tlp_cat_marc_field((subfields -> 0 ->> 'value'))
  WHERE tag = '245';

COMMENT ON TABLE tlp_cat_marc_field IS
  'KORMARC/MARC21 가변·반복 필드. 제어필드(00X)는 control_value, 데이터필드는 subfields(JSONB).';
COMMENT ON COLUMN tlp_cat_marc_field.subfields IS
  'JSONB array of {code:CHAR(1), value:TEXT}. 식별기호 단위 편집·반복 가능.';
```

DO $$
DECLARE
  t RECORD;
BEGIN
  FOR t IN
    SELECT table_name
    FROM information_schema.columns
    WHERE table_schema = 'tulip'
      AND column_name = 'tenant_id'
      AND table_name LIKE 'tlp_%'
  LOOP
    EXECUTE format('ALTER TABLE %I ENABLE ROW LEVEL SECURITY;', t.table_name);
    EXECUTE format($q$
      CREATE POLICY IF NOT EXISTS pol_%I_tenant_iso ON %I
      USING (tenant_id = fn_current_tenant_id() OR tenant_id = 0);
    $q$, t.table_name, t.table_name);
  END LOOP;
END $$;

## 10. 트리거·뷰 (선택)

```sql
-- 대출 시 자료 상태 자동 갱신 트리거
CREATE OR REPLACE FUNCTION fn_loan_after_insert()
RETURNS TRIGGER AS $$
BEGIN
  UPDATE tlp_col_copy
    SET item_status = 'ON_LOAN', updated_at = now()
    WHERE id = NEW.copy_id;
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_loan_ai AFTER INSERT ON tlp_cir_loan
  FOR EACH ROW EXECUTE FUNCTION fn_loan_after_insert();

-- 연체 자료 뷰
CREATE OR REPLACE VIEW vw_cir_overdue_loan AS
SELECT l.*, (CURRENT_DATE - l.due_date) AS overdue_days
FROM tlp_cir_loan l
WHERE l.status = 'ACTIVE'
  AND l.due_date < CURRENT_DATE;

-- 회원 대출현황 뷰
CREATE OR REPLACE VIEW vw_cir_member_loans AS
SELECT m.id AS member_id,
       m.member_no,
       COUNT(*) FILTER (WHERE l.status = 'ACTIVE') AS active_loans,
       COUNT(*) FILTER (WHERE l.status = 'OVERDUE') AS overdue_loans,
       COALESCE(SUM(f.amount - f.paid_amount - f.waived_amount)
         FILTER (WHERE f.status IN ('PENDING','PARTIAL')), 0) AS unpaid_fine
FROM tlp_cmn_member m
LEFT JOIN tlp_cir_loan l ON l.member_id = m.id
LEFT JOIN tlp_cir_fine f ON f.member_id = m.id
GROUP BY m.id, m.member_no;
```
